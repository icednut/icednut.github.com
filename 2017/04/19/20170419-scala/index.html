1val myArray = ArrayBuffer(3,6,7,3,-4,8,-3,-5,6,7,-2,9,-1)
2
3  def myRmNeg(targetArray: ArrayBuffer[Int]): ArrayBuffer[Int] = {
4      var index = 0
5      var limit = myArray.length
6      var isNegativeFound: Boolean = false
7      val newMyArray = ArrayBuffer[Int]()
8
9      while(index < limit) {
10        val num = targetArray(index)
11    
12        if (num < 0) {
13            if (!isNegativeFound) {
14                isNegativeFound = true
15                newMyArray += num
16            }
17        }
18        else newMyArray += num
19        
20        index += 1
21      }
22      newMyArray
23  }                                               //> myRmNeg: (targetArray: scala.collection.mutable.ArrayBuffer[Int])scala.colle
24                                                  //| ction.mutable.ArrayBuffer[Int]

26  def myRmNeg2(targetArray: ArrayBuffer[Int]): ArrayBuffer[Int] = {
27    val targetArraySize: Int = targetArray.length

30    var negativeIndexArray = for (index <- 0 until targetArraySize if targetArray(index) < 0) yield index
31        negativeIndexArray = negativeIndexArray.drop(1).reverse

Lab 4.3 Uniform Access (con’t)

  • 위 문제에서 uniform access method를 통해서 setter 구현해보기
  • 아래 코드를 보면 메소드명 바로 뒤에 ’_=‘가 붙는 것을 볼 수가 있는데 이것은 ‘변수명 =‘과 같은 효과를 발휘한다.
  • 또한 setter uniform access method를 구현하려면 앞에서 작성했던 getter에 해당하는 uniform access method가 먼저 선언되어 있어야 한다.

7
8    def hours_=(newHours: Int) {
9        if (newHours < 0 || newHours >= 24) {
10            throw new IllegalArgumentException("hours is invalid")
11        }
12        this.minutesSinceMidnight = newHours * 60 + this.minutes
13    }
14
15    def minutes_=(newMinutes: Int) {
16        if (newMinutes < 0 || newMinutes >= 60) {
17            throw new IllegalArgumentException("minutes is invalid")
18        }
19        this.minutesSinceMidnight = this.hours * 60 + newMinutes
20    }
21  }
